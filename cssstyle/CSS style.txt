    <py-script>
        from pyscript import Element

        def respond(event):
            user_input = Element('userInput').element.value.lower()
            if "halo" in user_input:
                response = "Halo! Apa yang bisa saya bantu?"
            elif "hai" in user_input:
                response ="haii juga"
            elif "apa kabar" in user_input:
                response = "Saya baik-baik saja, terima kasih!"
            elif"kamu siapa" in user_input:
                response = "saya adalah chatbot buatan Alma"
            elif"jelaskan tentang usu" in user_input:
                response = "USU adalah singkatan dari Universitas Sumatera Utara, yang beralamat di jl.Djamin Ginting"
            else:
                response = "Maaf, saya tidak mengerti."

            Element('chatOutput').element.innerHTML += f"<p><strong>Kamu:</strong> {user_input}</p>"
            Element('chatOutput').element.innerHTML += f"<p><strong>Bot:</strong> {response}</p>"
            Element('userInput').element.value = ""

        Element('sendBtn').element.onclick = respond
    </py-script>
